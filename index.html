            const handleKeyDown = (e) => { keysRef.current[e.code] = true; if (e.code === 'Space') fireBullet(); if (e.code === 'ShiftLeft' || e.code === 'KeyE') activateShield(); };
            const handleKeyUp = (e) => { keysRef.current[e.code] = false; };
            window.addEventListener('keydown', handleKeyDown); window.addEventListener('keyup', handleKeyUp);
           
            const canvas = canvasRef.current; const ctx = canvas.getContext('2d');
            const loop = (time) => {
                try {
                    if (stateRef.current.status === 'PLAYING') {
                        if (keysRef.current['ArrowLeft'] || keysRef.current['KeyA'] || touchLeftRef.current) playerRef.current.x = Math.max(0, playerRef.current.x - 8);
                        if (keysRef.current['ArrowRight'] || keysRef.current['KeyD'] || touchRightRef.current) playerRef.current.x = Math.min(CANVAS_WIDTH - 40, playerRef.current.x + 8);
                       
                        const actualIdx = stateRef.current.waveSequence[stateRef.current.wave];
                        const config = WAVE_CONFIGS[actualIdx];
                        let spawnRate = config.spawnRate / stateRef.current.difficultyMultiplier;
                        if (time - lastSpawnRef.current > spawnRate) {
                            if (!factBagsRef.current[config.type] || factBagsRef.current[config.type].length === 0) {
                                factBagsRef.current[config.type] = [...EM_FACTS[config.type]].sort(() => Math.random() - 0.5);
                            }
                            const fact = factBagsRef.current[config.type].pop();

                            invadersRef.current.push({ id: Math.random(), x: Math.random() * (CANVAS_WIDTH - 170), y: -80, width: 170, height: 80, fact, speedY: config.speed * stateRef.current.difficultyMultiplier, vx: 0, color: config.color, spawnTime: time, isLanding: false, waveType: config.type });
                            lastSpawnRef.current = time;
                        }
                        if (time - lastSaucerSpawnRef.current > 20000) {
                            const side = Math.random() > 0.5 ? 'left' : 'right';
                            saucersRef.current.push({ x: side === 'left' ? -60 : CANVAS_WIDTH, y: 100 + Math.random() * 100, speed: side === 'left' ? 3.5 : -3.5 });
                            lastSaucerSpawnRef.current = time;
                        }

                        for (let i = 0; i < invadersRef.current.length; i++) {
                            for (let j = i + 1; j < invadersRef.current.length; j++) {
                                const a = invadersRef.current[i]; const b = invadersRef.current[j];
                                if (!a.isLanding && !b.isLanding && Math.abs(a.x - b.x) < 160 && Math.abs(a.y - b.y) < 70) {
                                    const push = (a.x < b.x) ? -3 : 3; a.vx += push; b.vx -= push;
                                }
                            }
                        }

                        invadersRef.current.forEach(inv => {
                            if (inv.isLanding) {
                                inv.x += inv.vx;
                            } else {
                                const elapsed = (time - inv.spawnTime) / 1000;
                                switch(inv.waveType) {
                                    case 'Radio': inv.x += Math.sin(elapsed) * 1.5; break;
                                    case 'Microwave': inv.x += Math.sin(elapsed * 2) * 5; break;
                                    case 'Infrared': inv.x += Math.cos(elapsed) * 7; break;
                                    case 'Visible': inv.x += Math.sin(elapsed * 3) * 6; break;
                                    case 'Ultraviolet': inv.x += Math.sin(elapsed * 6) * 9; break;
                                    case 'X-Ray': if (Math.random() > 0.96) inv.vx = (Math.random() - 0.5) * 25; break;
                                    case 'Gamma': inv.x += (playerRef.current.x + 20 - (inv.x + 85)) * 0.015; break;
                                }
                                inv.y += (inv.speedY + (activeEffectsRef.current.timeWarp > 0 ? -0.2 : 0));
                                inv.x = Math.max(0, Math.min(CANVAS_WIDTH - 170, inv.x + inv.vx));
                                inv.vx *= 0.9;
                                if (inv.x < playerRef.current.x + 40 && inv.x + 170 > playerRef.current.x && inv.y < playerRef.current.y + 45 && inv.y + 80 > playerRef.current.y) {
                                    handleLifeLoss(); inv.y = 2000;
                                }
                            }
                        });

                        bulletsRef.current.forEach(b => b.y -= 12);
                        saucersRef.current.forEach(s => s.x += s.speed);
                        particlesRef.current.forEach(p => { p.x += p.vx; p.y += p.vy; p.life -= 0.03; });
                        particlesRef.current = particlesRef.current.filter(p => p.life > 0);

                        bulletsRef.current = bulletsRef.current.filter(b => {
                            let hit = false;
                            invadersRef.current = invadersRef.current.filter(inv => {
                                if (!inv.isLanding && b.x < inv.x + 170 && b.x + 4 > inv.x && b.y < inv.y + 80 && b.y + 15 > inv.y) {
                                    hit = true; createExplosion(inv.x + 85, inv.y + 40, inv.color);
                                    if (!inv.fact.isTrue) {
                                        setGameState(s => ({ ...s, score: s.score + 100 }));
                                        // RELEASE POWER UP RANDOMLY ON FALSE FACT HIT
                                        if (Math.random() > 0.8) {
                                            powerUpsRef.current.push({ id: Math.random(), x: inv.x + 70, y: inv.y + 25, type: Math.random() > 0.5 ? 'SHIELD' : 'TIME_WARP' });
                                        }
                                    } else handleLifeLoss(); return false;
                                } return true;
                            });
                            saucersRef.current = saucersRef.current.filter(s => {
                                if (b.x < s.x + 60 && b.x + 4 > s.x && b.y < s.y + 25 && b.y + 15 > s.y) {
                                    hit = true; createExplosion(s.x + 30, s.y + 12, '#f43f5e');
                                    setGameState(s => ({ ...s, score: s.score + 150 })); audioService.playPowerUp(); return false;
                                } return s.x > -100 && s.x < CANVAS_WIDTH + 100;
                            });
                            return !hit && b.y > 0;
                        });

                        powerUpsRef.current.forEach(p => p.y += 3);
                        powerUpsRef.current = powerUpsRef.current.filter(p => {
                            if (p.x < playerRef.current.x + 40 && p.x + 30 > playerRef.current.x && p.y < playerRef.current.y + 45 && p.y + 30 > playerRef.current.y) {
                                audioService.playPowerUp();
                                if (p.type === 'SHIELD') setGameState(s => ({ ...s, shieldTime: s.shieldTime + 5 }));
                                else activeEffectsRef.current.timeWarp = 400; return false;
                            } return p.y < CANVAS_HEIGHT;
                        });

                        invadersRef.current = invadersRef.current.filter(inv => {
                            if (!inv.isLanding && inv.y > CANVAS_HEIGHT - 50) {
                                inv.isLanding = true;
                                inv.vx = (inv.x + 85 < playerRef.current.x + 20) ? -22 : 22;
                                if (inv.fact.isTrue) setGameState(s => ({ ...s, score: s.score + 75 }));
                                else handleLifeLoss(true);
                            }
                            return inv.x > -250 && inv.x < CANVAS_WIDTH + 250 && inv.y < 1500;
                        });

                        if (stateRef.current.score >= (stateRef.current.wave + 1) * 600) {
                            const nextW = stateRef.current.wave + 1;
                            setGameState(s => ({ ...s, status: 'WAVE_TRANSITION', nextWaveTypeIndex: s.waveSequence[nextW >= 7 ? 0 : nextW] }));
                            setTimeout(() => {
                                setGameState(s => ({ ...s, wave: nextW >= 7 ? 0 : nextW, difficultyMultiplier: s.difficultyMultiplier * 1.2, status: 'PLAYING' }));
                                invadersRef.current = []; bulletsRef.current = [];
                            }, 2500);
                        }

                        ctx.clearRect(0, 0, CANVAS_WIDTH, CANVAS_HEIGHT);
                        ctx.fillStyle = 'white'; for (let i = 0; i < 20; i++) ctx.fillRect((Math.sin(i * 123) * 0.5 + 0.5) * CANVAS_WIDTH, (Math.cos(i * 678 + time/1500) * 0.5 + 0.5) * CANVAS_HEIGHT, 1, 1);
                       
                        drawMoon(ctx);
                        drawPlayer(ctx);
                        invadersRef.current.forEach(inv => {
                            drawAlien(ctx, inv);
                            ctx.fillStyle = 'white'; ctx.font = 'bold 14px Inter'; ctx.textAlign = 'center';
                            const words = inv.fact.text.split(' '); let line = ''; let ly = inv.y + 40;
                            words.forEach(w => { if (ctx.measureText(line + w).width > 150) { ctx.fillText(line, inv.x+85, ly); line = w + ' '; ly += 16; } else line += w + ' '; });
                            ctx.fillText(line, inv.x+85, ly);
                        });
                        bulletsRef.current.forEach(b => { ctx.fillStyle = '#fbbf24'; ctx.fillRect(b.x, b.y, 4, 15); });
                        saucersRef.current.forEach(s => drawSaucer(ctx, s));
                        powerUpsRef.current.forEach(p => {
                            const rot = time / 500; ctx.save(); ctx.translate(p.x+15, p.y+15); ctx.rotate(rot);
                            ctx.fillStyle = p.type === 'SHIELD' ? '#3b82f6' : '#a855f7';
                            ctx.beginPath(); ctx.arc(0, 0, 15, 0, Math.PI*2); ctx.fill();
                            ctx.strokeStyle = 'white'; ctx.lineWidth = 2; ctx.stroke();
                            ctx.fillStyle = 'white'; ctx.font = 'bold 14px Inter'; ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
                            ctx.fillText(p.type === 'SHIELD' ? 'S' : 'T', 0, 0); ctx.restore();
                        });
                        particlesRef.current.forEach(p => { ctx.globalAlpha = p.life; ctx.fillStyle = p.color; ctx.fillRect(p.x, p.y, 3, 3); ctx.globalAlpha = 1; });
                       
                        if (activeEffectsRef.current.shield > 0) activeEffectsRef.current.shield--;
                        if (activeEffectsRef.current.timeWarp > 0) activeEffectsRef.current.timeWarp--;
                    }
                    loopRef.current = requestAnimationFrame(loop);
                } catch (e) { loopRef.current = requestAnimationFrame(loop); }
            };
            loopRef.current = requestAnimationFrame(loop);
            return () => {
                window.removeEventListener('keydown', handleKeyDown); window.removeEventListener('keyup', handleKeyUp);
                cancelAnimationFrame(loopRef.current);
                isLoopRunning.current = false;
            };
        }, [fireBullet, activateShield]);

        return (
            <div className="game-wrapper">
                <div className="canvas-container">
                    <canvas ref={canvasRef} width={CANVAS_WIDTH} height={CANVAS_HEIGHT} />
                   
                    <div className="absolute top-0 left-0 right-0 p-4 flex justify-between items-start pointer-events-none z-40">
                        <div className="flex flex-col gap-2">
                            <div className="bg-black/70 px-4 py-2 rounded-full border border-white/20 flex items-center gap-3 shadow-2xl">
                                <Icons.Trophy /><span className="font-black text-2xl text-white">{gameState.score.toLocaleString()}</span>
                            </div>
                            <div className="flex gap-1.5">{[...Array(5)].map((_, i) => <Icons.Heart key={i} filled={i < gameState.lives} />)}</div>
                        </div>
                        <div className="bg-black/70 px-6 py-4 rounded-2xl border border-white/20 text-right shadow-2xl backdrop-blur-md">
                            <div className="text-3xl font-black uppercase italic leading-none mb-1" style={{ color: WAVE_CONFIGS[gameState.waveSequence[gameState.wave]]?.color }}>
                                {WAVE_CONFIGS[gameState.waveSequence[gameState.wave]]?.type}
                            </div>
                            <div className="text-sm text-zinc-300 font-black uppercase tracking-widest">Wave {gameState.wave + 1} / 7</div>
                        </div>
                    </div>

                    {gameState.status === 'START' && (
                        <div className="menu-overlay">
                            <div className="bg-zinc-900 border-4 border-emerald-500 p-8 rounded-2xl w-full max-h-full overflow-y-auto custom-scrollbar">
                                <h1 className="text-5xl font-black text-emerald-500 mb-6 italic uppercase tracking-tighter">EM Invaders</h1>
                                <div className="bg-emerald-500/10 border border-emerald-500/30 p-6 rounded-xl mb-6">
                                    <h2 className="text-emerald-400 font-black uppercase text-xl mb-3 underline decoration-2 underline-offset-4">Mission Briefing</h2>
                                    <p className="text-zinc-100 text-lg leading-relaxed font-bold">
                                        <span className="text-emerald-400 font-black uppercase">Shoot FAKE facts</span> about each EM wave!<br/>
                                        <span className="text-blue-400 font-black uppercase">Protect TRUE facts</span> by letting them land safely.<br/>
                                        <span className="text-yellow-400 font-black uppercase">Tactical Tip:</span> Power-ups drop from destroyed FAKE facts!
                                    </p>
                                </div>
                                <div className="space-y-4 mb-8">
                                    <div className="bg-zinc-800 p-5 rounded-xl border border-zinc-700 text-left">
                                        <h3 className="text-sm text-zinc-500 uppercase font-black mb-2">Keyboard Command</h3>
                                        <p className="text-base text-white font-bold">Arrows: Move Ship • Space: Fire Blaster • Shift: Shield</p>
                                    </div>
                                    <div className="bg-zinc-800 p-5 rounded-xl border border-zinc-700 text-left">
                                        <h3 className="text-sm text-zinc-500 uppercase font-black mb-2">Touch Tactical</h3>
                                        <p className="text-base text-white font-bold">Use the large tactical buttons at the bottom of your screen.</p>
                                    </div>
                                </div>
                                <button onClick={initGame} className="btn-primary">Initialize Defense</button>
                            </div>
                        </div>
                    )}

                    {gameState.status === 'GAME_OVER' && (
                        <div className="menu-overlay">
                            <div className="bg-zinc-900 border-4 border-red-500 p-8 rounded-2xl w-full">
                                <h1 className="text-4xl font-black text-red-500 mb-4 uppercase italic">Mission Failed</h1>
                                <div className="text-3xl font-bold mb-8 text-white">Score: {gameState.score.toLocaleString()}</div>
                                <button onClick={() => setGameState(s => ({ ...s, status: 'INPUT_INITIALS' }))} className="btn-primary mb-4 bg-yellow-500">Save Score</button>
                                <button onClick={initGame} className="btn-primary bg-red-600 text-white">Try Again</button>
                            </div>
                        </div>
                    )}

                    {gameState.status === 'INPUT_INITIALS' && (
                        <div className="menu-overlay">
                            <div className="bg-zinc-900 border-4 border-yellow-500 p-8 rounded-2xl w-full">
                                <h1 className="text-3xl font-black text-yellow-500 mb-6 uppercase">New High Score!</h1>
                                <input maxLength={3} value={initials} onChange={e => setInitials(e.target.value.toUpperCase())} className="w-full bg-zinc-800 p-5 rounded-xl text-center text-5xl font-black mb-6 uppercase border-2 border-yellow-500/50" placeholder="AAA" />
                                <button onClick={handleSaveScore} disabled={!initials || isSaving} className="btn-primary bg-yellow-500">{isSaving ? 'Saving...' : 'Confirm'}</button>
                                <button onClick={fetchLeaderboard} className="text-sm text-zinc-500 mt-6 uppercase font-black hover:text-zinc-300">Skip to Leaderboard</button>
                            </div>
                        </div>
                    )}

                    {gameState.status === 'WAVE_TRANSITION' && (
                        <div className="menu-overlay">
                            <div className="text-center">
                                <h1 className="text-5xl font-black mb-4 uppercase italic text-white">Wave Clear</h1>
                                <div className="text-4xl font-black animate-pulse uppercase mb-4" style={{ color: WAVE_CONFIGS[gameState.nextWaveTypeIndex]?.color }}>
                                    {WAVE_CONFIGS[gameState.nextWaveTypeIndex]?.type} Incoming
                                </div>
                                <div className="text-xl text-zinc-400 uppercase tracking-widest font-black">Wave {gameState.wave + 2} / 7</div>
                            </div>
                        </div>
                    )}

                    {showLeaderboard && (
                        <div className="menu-overlay">
                            <div className="bg-zinc-900 border-4 border-yellow-500 p-6 rounded-2xl w-full">
                                <h1 className="text-2xl font-black mb-6 uppercase italic text-white">Global Top 10 Defenders</h1>
                                <div className="space-y-2 mb-8 max-h-[350px] overflow-y-auto pr-2 custom-scrollbar">
                                    {leaderboardScores.map((s, i) => (
                                        <div key={s.id} className="flex justify-between p-3 bg-zinc-800 rounded-xl border border-zinc-700">
                                            <span className="font-black text-lg text-white">{i+1}. {s.initials}</span>
                                            <span className="text-yellow-500 font-black text-lg">{s.score.toLocaleString()}</span>
                                        </div>
                                    ))}
                                </div>
                                <button onClick={initGame} className="btn-primary bg-yellow-500">Play Again</button>
                            </div>
                        </div>
                    )}
                </div>
               
                <div className="w-full p-4 flex items-center justify-between gap-4 select-none">
                    <div className="flex gap-4">
                        <button onMouseDown={() => touchLeftRef.current = true} onMouseUp={() => touchLeftRef.current = false} onTouchStart={(e) => { e.preventDefault(); touchLeftRef.current = true; }} onTouchEnd={(e) => { e.preventDefault(); touchLeftRef.current = false; }} className="control-btn w-16 h-16 bg-zinc-800 border-4 border-zinc-700 rounded-2xl flex items-center justify-center active:bg-zinc-700 shadow-lg"><Icons.ArrowLeft /></button>
                        <button onMouseDown={() => touchRightRef.current = true} onMouseUp={() => touchRightRef.current = false} onTouchStart={(e) => { e.preventDefault(); touchRightRef.current = true; }} onTouchEnd={(e) => { e.preventDefault(); touchRightRef.current = false; }} className="control-btn w-16 h-16 bg-zinc-800 border-4 border-zinc-700 rounded-2xl flex items-center justify-center active:bg-zinc-700 shadow-lg"><Icons.ArrowRight /></button>
                    </div>
                    <div className="flex gap-4 items-center">
                        <button onClick={activateShield} disabled={gameState.shieldTime <= 0} className={`control-btn w-16 h-16 rounded-2xl flex items-center justify-center border-4 shadow-lg transition-all ${gameState.shieldTime > 0 ? 'bg-blue-600 border-blue-400 active:bg-blue-500' : 'bg-zinc-800 border-zinc-700 opacity-30 grayscale'}`}><Icons.Zap filled={gameState.shieldTime > 0} /></button>
                        <button onClick={fireBullet} className="shoot-btn w-20 h-20 bg-emerald-600 border-4 border-emerald-400 rounded-full flex items-center justify-center active:bg-emerald-500 shadow-[0_0_30px_rgba(16,185,129,0.4)]"><Icons.Zap filled={true} /></button>
                    </div>
                </div>
            </div>
        );
    };

    ReactDOM.createRoot(document.getElementById('root')).render(<Game />);
</script>
